\begin{frame}[fragile]
\frametitle{Well-formedness}
% WF-MI
\begin{prooftree}
\AxiomC{\FVeq{\ovl{a}}{x}}
\AxiomC{\FVeq{\ovl{b}}{x,y}}
\AxiomC{\typeass{\ovl{a}}{e}{\type{y}{\ovl{b}}}}
\RightLabel{WF-MI}
\TrinaryInfC{\wf{(\mImpl{m}{x}{\ovl{a}}{\type{y}{\ovl{b}}}{e})}}
\end{prooftree}
\pause
\onslide<+->
\vfill
\begin{lstlisting}[basicstyle=\footnotesize]
def wf(D: Declaration): Boolean = D match {
  case ...
  case MethodImplementation(_, x, a, Type(y, b), e) =>
    FV(a) == List(x) && FV(b).nonEmpty &&
      FV(b).forall(v => v == x || v == y) &&
    typeassignment(a, e).exists {
      case Type(z, c) =>
        c.size == b.size &&
        substitute(z, y, c).forall(b.contains(_))
    }
}
\end{lstlisting}
\end{frame}
